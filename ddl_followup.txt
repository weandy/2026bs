CREATE TABLE `follow_up_plan` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `resident_id` bigint NOT NULL COMMENT '关联居民（跨库引用）',
  `resident_name` varchar(20) NOT NULL COMMENT '冗余，避免跨库查询',
  `chronic_type` varchar(30) NOT NULL COMMENT '慢病类型（hypertension/diabetes/chd等）',
  `doctor_id` bigint NOT NULL COMMENT '责任医生',
  `doctor_name` varchar(20) NOT NULL,
  `frequency` tinyint(1) NOT NULL COMMENT '随访频次：1每月 2每季度 3每半年',
  `follow_up_method` tinyint(1) NOT NULL DEFAULT '1' COMMENT '1电话 2门诊 3上门',
  `next_follow_date` date NOT NULL COMMENT '下次随访日期',
  `status` tinyint(1) NOT NULL DEFAULT '1' COMMENT '1管理中 2已停止',
  `created_by` bigint NOT NULL,
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `is_deleted` tinyint(1) NOT NULL DEFAULT '0',
  `deleted_at` datetime DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `idx_resident_id` (`resident_id`),
  KEY `idx_doctor_id` (`doctor_id`),
  KEY `idx_next_date` (`next_follow_date`),
  KEY `idx_chronic_type` (`chronic_type`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='慢病随访计划'

CREATE TABLE `follow_up_record` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `plan_id` bigint NOT NULL,
  `resident_id` bigint NOT NULL,
  `follow_date` date NOT NULL COMMENT '实际随访日期',
  `follow_method` tinyint(1) NOT NULL,
  `systolic_bp` smallint DEFAULT NULL COMMENT '收缩压（mmHg）',
  `diastolic_bp` smallint DEFAULT NULL COMMENT '舒张压（mmHg）',
  `fasting_glucose` decimal(5,2) DEFAULT NULL COMMENT '空腹血糖（mmol/L）',
  `postprandial_glucose` decimal(5,2) DEFAULT NULL COMMENT '餐后2h血糖',
  `medication_compliance` tinyint(1) DEFAULT NULL COMMENT '用药依从性：1规律 2间断 3不服药',
  `lifestyle_issues` varchar(200) DEFAULT NULL COMMENT '生活方式问题（JSON数组）',
  `health_guidance` text COMMENT '健康指导内容',
  `next_follow_date` date DEFAULT NULL,
  `staff_id` bigint NOT NULL,
  `staff_name` varchar(20) NOT NULL,
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `idx_plan_id` (`plan_id`),
  KEY `idx_resident_id` (`resident_id`),
  KEY `idx_follow_date` (`follow_date`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='慢病随访记录'

CREATE TABLE `public_health_record` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `resident_id` bigint NOT NULL,
  `resident_name` varchar(20) NOT NULL,
  `service_type` varchar(30) NOT NULL COMMENT '服务类型：elder_exam/prenatal/child_care',
  `service_date` date NOT NULL,
  `indicators` json NOT NULL COMMENT '检查指标（JSON，各服务类型结构不同）',
  `conclusion` tinyint(1) DEFAULT NULL COMMENT '结论：1正常 2异常',
  `conclusion_desc` varchar(500) DEFAULT NULL,
  `referral_needed` tinyint(1) NOT NULL DEFAULT '0' COMMENT '是否建议转诊',
  `referral_reason` varchar(200) DEFAULT NULL,
  `staff_id` bigint NOT NULL,
  `staff_name` varchar(20) NOT NULL,
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `is_deleted` tinyint(1) NOT NULL DEFAULT '0',
  `deleted_at` datetime DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `idx_resident_service` (`resident_id`,`service_type`),
  KEY `idx_service_date` (`service_date`),
  KEY `idx_service_type` (`service_type`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='公卫服务记录'
